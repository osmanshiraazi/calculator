```javascript
const buttons = document.querySelectorAll('button');

buttons.forEach((button) => {
  button.addEventListener('click', (e) => {
    const input = document.querySelector('input');
    const value = e.target.innerHTML;

    if (value === '=') {
      input.value = eval(input.value);
    } else if (value === 'C') {
      input.value = '';
    } else {
      input.value += value;
    }
  });
});
